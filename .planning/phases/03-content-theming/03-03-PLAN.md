```typescript
const animateTransition = contextSafe(
  (newIndex: number, prevIndex: number, dir: number) => {
    const tl = gsap.timeline({
      onComplete: () => useSliderStore.getState().setAnimating(false),
    });
    tl.to(`[data-index="${prevIndex}"]`, { xPercent: -100 * dir, duration: 0.8, ease: 'power3.inOut' });
    tl.fromTo(`[data-index="${newIndex}"]`, { xPercent: 100 * dir }, { xPercent: 0, duration: 0.8, ease: 'power3.inOut' }, '<');
  }
);
```
