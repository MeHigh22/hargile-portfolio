**SceneRenderer.tsx:**
Single component that dispatches by scene prop:
```tsx
import { useId } from 'react';
import type { SceneKind } from '../types';

export function SceneRenderer({ scene }: { scene: SceneKind }) {
  const uid = useId();
  switch (scene) {
    case 'dashboard': return <DashboardScene uid={uid} />;
    case 'trading':   return <TradingScene uid={uid} />;
    case 'gallery':   return <GalleryScene uid={uid} />;
    case 'banking':   return <BankingScene uid={uid} />;
    case 'editorial': return <EditorialScene uid={uid} />;
    case 'shop':      return <ShopScene uid={uid} />;
  }
}
```

Each scene sub-component (`DashboardScene`, `TradingScene`, `GalleryScene`, `BankingScene`, `EditorialScene`, `ShopScene`) lives in the same file, takes `{ uid: string }` props, and uses `${uid}-` prefix on all SVG id attributes. Port the SVG content from the corresponding scene function in `public/portfolio/index.html` (look for `function dashboard()`, `function trading()`, etc. in the `<script>` block). Convert innerHTML strings → JSX SVG. Replace all `id="af"` style attributes with `id={\`${uid}-af\`}` and all `fill="url(#af)"` with `fill={\`url(#${uid}-af)\`}`.
